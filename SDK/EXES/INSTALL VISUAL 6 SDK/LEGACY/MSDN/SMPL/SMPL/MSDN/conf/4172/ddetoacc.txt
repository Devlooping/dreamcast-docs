Sub MAIN
' Using the System topic, open the ac203.MDB database.
' The database must be open before you can use other DDE topics.
' You must update the OpenDatabase statement to use a valid path

Chan1 = DDEInitiate("MSAccess", "System")
DDEExecute Chan1, "[OpenDatabase d:\access95\te95\ac203\ac203.MDB]"

' Get all the data from the Customer List query.

Chan2 = DDEInitiate("MSAccess", "ac203;QUERY Customer List")
MyData$ = DDERequest$(Chan2, "All")
DDETerminate Chan2

' Close the database.

DDEExecute Chan1, "[CloseDatabase]"
DDETerminate Chan1

Insert MyData$

End Sub
